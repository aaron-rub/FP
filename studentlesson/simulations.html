cor = 0.7

bounces = 10

for i in range(bounces):
    #calculate the height of the ball on this bounce
    height = height * cor
    #check if the ball is still above 0.01 units
    if height < 0.01:
        #end the simulation
        break
    #output the height of the ball on this bounce
    print("Bounce", i+1, ": height =", height)
